28  /**
29   * Manager for displaying paginated data.
30   * 
31   * <p>Note that, in order for an instance of this class to begin operating, it must be registered
32   * with a {@link ButtonManager} using the handler given by {@link #pageUpdateHandler()}.

39      /** The base ID for an update button. */
40      public static final String BUTTON_ID_UPDATE = "page-update";
41      /** The ID for the counter display button. */
42      public static final String BUTTON_ID_DISPLAY = "page-display";

44      /** The label for the previous page button. */
45      private static final ReactionEmoji BUTTON_LABEL_PREVIOUS = ReactionEmoji.unicode( "‚óÄÔ∏è" );
46      /** The label for the next page button. */

48      /** The label for the refresh page button. */
49      private static final ReactionEmoji BUTTON_LABEL_REFRESH = ReactionEmoji.unicode( "üîÅ" );

51      /** Separator character for arguments. */
52      private static final String ARG_SEPARATOR = ":";
53  
54      /** Logger. */
55      private static final Logger LOGGER = LoggerFactory.getLogger( PaginationManager.class );
56  
57      /** The registered paginators. */
58      private final Map<String, Paginator> paginators = new HashMap<>();
59  
60      /** Creates a new instance. */
61      public PaginationManager() {}

63      /**
64       * Encodes page parameters into a button arg.
65       *
66       * @param id The page type ID.
67       * @param index The page index.
68       * @param args The page arguments. An empty string is treated as no arguments.
69       * @return The encoded parameters.
70       */
71      private String encode( 
72              final String id,
73              final @NonNegative int index, 
74              final String args 
75      ) {
76  
77          final var pageStr = String.valueOf( index );
78          return args.isEmpty() 
79                  ? String.join( ARG_SEPARATOR, id, pageStr )
80                  : String.join( ARG_SEPARATOR, id, pageStr, args );
81  
82      }
83  
84      /**
85       * Decodes page parameters.
86       *
87       * @param arg The parameters encoded by {@link #encode(String, int, String)}.
88       * @return The decoded parameters, in the order taken by {@link #encode(String, int, String)}.
89       * @throws IllegalArgumentException If the given arg is not valid.
90       */
91      private Tuple3<String, @NonNegative Integer, String> decode( final String arg ) {
92  
93          final var s = arg.split( ARG_SEPARATOR, 3 );
94          if ( s.length < 2 ) {
95              throw new IllegalArgumentException( "Not a valid arg: " + arg );
96          }
97  
98          final var id = s[0];
99          final var index = Integer.parseInt( s[1] );
100         final var args = s.length > 2 ? s[2] : "";
101 
102         if ( index < 0 ) {
103             throw new IllegalArgumentException( "Arg has negative index" );
104         }
105 
106         return Tuples.of( id, index, args );
107 
108     }

110     /**
111      * Registers a paginator.
112      *
113      * @param id The page type ID.
114      * @param requiredGroup The group that a user must belong to to interact with
115      *                      the page controls. If {@code null}, defaults to everyone.
116      * @param generator The generator function for page data.
117      * @return The paginator.
118      * @throws IllegalArgumentException if there is already a registered paginator
119      *                                  for the given type ID.
120      */
121     public synchronized Paginator register( 
122             final String id, 
123             final @Nullable Group requiredGroup,
124             final PageGenerator generator 
125     ) throws IllegalArgumentException {
126 
127         if ( paginators.containsKey( id ) ) {
128             throw new IllegalArgumentException( "ID already in use" );
129         }
130 
131         LOGGER.info( "Registering paginator {}", id );
132 
133         final var paginator = new Paginator( 
134                 generator, 
135                 id, 
136                 Objects.requireNonNullElse( requiredGroup, Groups.EVERYONE )
137         );
138         
139         paginators.put( id, paginator );
140         return paginator;
141 
142     }

153      * @return The paginator.
154      * @throws IllegalArgumentException if there is already a registered paginator
155      *                                  for the given type ID.
156      */
157     public Paginator register( 
158             final String id,
159             final PageGenerator generator

166     /**
167      * Configures the handler for page update buttons. Must be registered with a
168      * button manager for this pagination manager to work.
169      *
170      * @return The handler.
171      */
172     public ButtonManager.Handler pageUpdateHandler() {
173 
174         return ButtonManager.Handler.builder().handler( ( ctx, arg ) -> {
175 
176             final var decoded = decode( arg );
177             final String id = decoded.getT1();
178             final int page = decoded.getT2();
179             final String args = decoded.getT3();
180 
181             // Get associated paginator
182             final var paginator = paginators.get( id );
183             if ( paginator == null ) {
184                 return ctx.event()
185                         .reply( "Page type not recognized" )
186                         .withEphemeral( true );
187             }
188 
189             // Check membership to required group
190             return paginator.requiredGroup.belongs( ctx )
191                     .flatMap( b -> {
192 
193                         final var event = ctx.event();
194 
195                         if ( b ) {
196                             // Membership validated, update page
197                             return paginator.generatePage( page, args )
198                                     .flatMap( event::edit );
199                         } else if ( paginator.requiredGroup instanceof NamedGroup g ) {
200                             // Not member, indicate group name
201                             return event.reply( String.format(
202                                     "You must be part of the %s group to change the page.",
203                                     g.name()
204                             ) );
205                         } else {
206                             // Not member, no group name available
207                             return event.reply( "You do not have permissions to change the page." );
208                         }
209 
210                     } );
211 
212         } )
213         .id( BUTTON_ID_UPDATE )
214         .mutex( true )
215         .build();
216 
217     }

219     /**
220      * The data of a page to be displayed.
221      *
222      * @param args The page arguments.
223      * @param index The page index.
224      * @param pageCount The total amount of pages available.
225      * @param content The page content.
226      * @param components The components to attach to the page. Only up to {@value #MAX_ROWS} 
227      *                   rows as one row is used by the page controls.
228      * @since 1.0
229      */
230     public record PageData(
231             String args,
232             @NonNegative int index,
233             @NonNegative int pageCount,
234             EmbedCreateSpec content,
235             List<LayoutComponent> components
236     ) {
237 
238         /** Maximum number of component rows. */
239         public static final int MAX_ROWS = 4;
240 
241         /**
242          * Creates a new instance.
243          *
244          * @throws IllegalArgumentException if more than {@value #MAX_ROWS} 
245          *                                  component rows are given.
246          */
247         public PageData {
248 
249             if ( components.size() > MAX_ROWS ) {
250                 throw new IllegalArgumentException( "Too many component rows" );
251             }
252 
253         }
254

319     /**
320      * Entity responsible for generating content pages of a certain type.
321      *
322      * @since 1.0
323      */
324     public final class Paginator {
325 
326         /** The generator to use for page data. */
327         private final PageGenerator generator;
328 
329         /** The page type ID. */
330         private final String id;
331 
332         /** The required group for using page controls. */
333         private final Group requiredGroup;
334 
335         /**
336          * Creates a new instance.
337          *
338          * @param generator The generator to use for page data.
339          * @param id The page type ID.
340          * @param requiredGroup The required group for using page controls.
341          */
342         private Paginator( 
343                 final PageGenerator generator, 
344                 final String id, 
345                 final Group requiredGroup 
346         ) {
347 
348             this.generator = generator;
349             this.id = id;
350             this.requiredGroup = requiredGroup;
351 
352         }
353 
354         /**
355          * Renders a page.
356          *
357          * @param data The page data.
358          * @return The rendered page.
359          */
360         private InteractionApplicationCommandCallbackSpec renderPage( final PageData data ) {
361 
362             LOGGER.trace( "Rendering page {}/{}/{}", id, data.args(), data.index() );
363 
364             final var buttonPrev = data.index() <= 0 // If first page
365                     ? Button.primary( BUTTON_ID_UPDATE + "-", BUTTON_LABEL_PREVIOUS ).disabled()
366                     : Button.primary( 
367                             ButtonManager.makeId( 
368                                     BUTTON_ID_UPDATE, 
369                                     encode( id, data.index() - 1, data.args() ) 
370                             ), 
371                             BUTTON_LABEL_PREVIOUS 
372                     );
373             final var buttonNext = data.index() >= data.pageCount() - 1 // If last page
374                     ? Button.primary( BUTTON_ID_UPDATE + "+", BUTTON_LABEL_NEXT ).disabled()
375                     : Button.primary( 
376                             ButtonManager.makeId( 
377                                     BUTTON_ID_UPDATE, 
378                                     encode( id, data.index() + 1, data.args() ) 
379                             ), 
380                             BUTTON_LABEL_NEXT 
381                     );
382 
383             final var navBar = ActionRow.of(
384                     buttonPrev,
385                     Button.secondary( 
386                             ButtonManager.makeId( 
387                                     BUTTON_ID_UPDATE, 
388                                     encode( id, data.index(), data.args() ) 
389                             ), 
390                             BUTTON_LABEL_REFRESH
391                     ),
392                     Button.secondary( 
393                             BUTTON_ID_DISPLAY, 
394                             "%d/%d".formatted( data.index() + 1, data.pageCount() ) 
395                     ).disabled(),
396                     buttonNext
397             );
398 
399             return InteractionApplicationCommandCallbackSpec.builder()
400                     .addEmbed( data.content() )
401                     .addComponent( navBar )
402                     .addAllComponents( data.components() )
403                     .ephemeral( false )
404                     .build();
405 
406         }
407 
408         /**
409          * Generates a page.
410          *
411          * @param index The page index.
412          * @param args The page arguments.
413          * @return The generated page.
414          * @throws IllegalArgumentException if the given index is negative.
415          * 
416          */
417         private Mono<InteractionApplicationCommandCallbackSpec> generatePage( 
418                 final @NonNegative int index, final String args 
419         ) throws IllegalArgumentException {
420 
421             if ( index < 0 ) {
422                 throw new IllegalArgumentException( "Negative indices not allowed" );
423             }
424 
425             return generator.generatePage( index, args )
426                     .map( this::renderPage );
427 
428         }
429 
430         /**
431          * Generates the first page.
432          * 
433          * <p>Equivalent to {@link #generatePage(int, String) generatePage(0, args)}.
434          *
435          * @param args The page arguments.
436          * @return The generated page.
437          */
438         public Mono<InteractionApplicationCommandCallbackSpec> generateFirst( final String args ) {
439 
440             return generatePage( 0, args );
441 
442         }
443 
444     }
