20  /**
21   * Reply manager for message-based commands.
22   *
23   * @version 1.0
24   * @since 1.0
25   */
26  public class MessageReplyManager extends AbstractReplyManager {
27  
28      /** Where to send public replies. */
29      private final ReplyChain publicChain;
30      /** 
31       * Where to send private replies. May be the same as {@link #publicChain} if the command
32       * is called from a private channel. 
33       */
34      private final ReplyChain privateChain;
35  
36      /**
37       * Creates a new manager.
38       *
39       * @param replies The existing replies.
40       * @param publicChain The reply chain for public replies.
41       * @param privateChain The reply chain for private replies. 
42       *                     If {@code null}, the public chain is also used for private 
43       *                     replies (so public and private replies are the same).
44       * @param defaultPrivate Whether replies are private by default.
45       */
46      @SideEffectFree
47      private MessageReplyManager( 
48              final List<? extends Reply> replies,
49              final ReplyChain publicChain,
50              final @Nullable ReplyChain privateChain,
51              final boolean defaultPrivate 
52      ) {
53  
54          super( replies, defaultPrivate );
55  
56          this.publicChain = publicChain;
57          this.privateChain = privateChain != null ? privateChain : publicChain;
58  
59      }
60  
61      /**
62       * Creates a new manager.
63       *
64       * @param replies The existing replies.
65       * @param referenceFormatter The formatter to use for the reference message.
66       * @param previous The ID of the latest message in the public reply chain.
67       * @param publicChannel The public channel to send messages in.
68       * @param privateChannel The private channel to send messages in.
69       *                       If {@code null}, the public channel is also used for private
70       *                       replies (so public and private replies are the same).
71       * @param defaultPrivate Whether replies are private by default.
72       */
73      @SideEffectFree
74      MessageReplyManager( 
75              final List<? extends Reply> replies,
76              final Supplier<EmbedCreateSpec> referenceFormatter,
77              final Mono<Snowflake> previous,
78              final Mono<MessageChannel> publicChannel, 
79              final @Nullable Mono<PrivateChannel> privateChannel,
80              final boolean defaultPrivate 
81      ) {
82  
83          this( 
84                  replies, 
85                  new ReplyChain( referenceFormatter, publicChannel, previous ),
86                  privateChannel != null
87                          ? new ReplyChain( referenceFormatter, privateChannel, Mono.empty() )
88                          : null,
89                  defaultPrivate
90          );
91  
92      }
93  
94      /**
95       * Creates a new manager.
96       *
97       * @param referenceFormatter The formatter to use for the reference message.
98       * @param original The original message that triggered the event.
99       * @param publicChannel The public channel to send messages in.
100      * @param privateChannel The private channel to send messages in.
101      *                       If {@code null}, the public channel is also used for private
102      *                       replies (so public and private replies are the same).
103      * @param defaultPrivate Whether replies are private by default.
104      */
105     @SideEffectFree
106     public MessageReplyManager( 
107             final Supplier<EmbedCreateSpec> referenceFormatter,
108             final Message original, 
109             final Mono<MessageChannel> publicChannel, 
110             final @Nullable Mono<PrivateChannel> privateChannel,
111             final boolean defaultPrivate
112     ) {
113 
114         this(
115                 Collections.emptyList(),
116                 referenceFormatter,
117                 Mono.just( original.getId() ),
118                 publicChannel,
119                 privateChannel,
120                 defaultPrivate
121         );
122 
123     }
124 
125     @Override
126     public Mono<Void> doDefer() {
127 
128         return ( defaultPrivate ? privateChain : publicChain ).channel
129                 .flatMap( MessageChannel::type );
130 
131     }
132 
133     @Override
134     protected Mono<Reply> send( final @NonNegative int index, final ReplySpec spec ) {
135 
136         final var channel = spec.privatelyOrElse( defaultPrivate ) ? privateChain : publicChain;
137         return channel.send( index, spec );
138 
139     }
140 
141     @Override
142     @SuppressWarnings( "interning:not.interned" ) // Intentional
143     public ReplyManager doDetach() {
144 
145         return new MessageReplyManager(
146                 replies, 
147                 publicChain.copy(), 
148                 privateChain == publicChain ? null : privateChain.copy(), 
149                 defaultPrivate
150         );
151 
152     }

154     /**
155      * The base for a message-based reply.
156      *
157      * @since 1.0
158      */
159     interface MessageReplyBase extends Reply {
160 
161         @Override
162         default Mono<Message> edit( final ReplyEditSpec spec ) {
163 
164             return message().flatMap( m -> m.edit( spec.toMessage() ) );
165 
166         }
167 
168         @Override
169         default Mono<Void> delete() {
170 
171             return message().flatMap( Message::delete );
172 
173         }
174 
175     }
176 
177     /**
178      * A reply made by the manager.
179      *
180      * @param index The reply index.
181      * @param id The ID of the reply message.
182      * @param channel The ID of the channel the reply was sent in.
183      * @param client The client used to connect to Discord.
184      * @since 1.0
185      */
186     record MessageReply( 
187             @NonNegative int index,
188             Snowflake id,
189             Snowflake channel,
190             GatewayDiscordClient client
191     ) implements MessageReplyBase {
192 
193         @Override
194         public Mono<Message> message() {
195 
196             return client.getMessageById( channel, id );
197 
198         }
199 
200         @Override
201         public Mono<Snowflake> messageId() {
202 
203             return Mono.just( id );
204 
205         }
206 
207     }

209     /**
210      * Manages a sequence of replies sent by the command.
211      *
212      * @since 1.0
213      */
214     private static class ReplyChain {
215 
216         /** The target channel. */
217         public final Mono<? extends MessageChannel> channel;
218 
219         /** The formatter for the reference message. */
220         private final Supplier<EmbedCreateSpec> referenceFormatter;
221         
222         /** The last sent reply ID, or {@code null} if none were sent in this channel yet. */
223         private Mono<Snowflake> last;
224 
225         /**
226          * Creates a new instance.
227          *
228          * @param referenceFormatter The formatter to use for the reference message.
229          * @param channel The target channel.
230          * @param last The last sent reply ID in this channel, or {@code null} 
231          *             if none were sent in this channel yet.
232          */
233         ReplyChain( 
234                 final Supplier<EmbedCreateSpec> referenceFormatter,
235                 final Mono<? extends MessageChannel> channel, 
236                 final Mono<Snowflake> last 
237         ) {
238 
239             this.referenceFormatter = Objects.requireNonNull( referenceFormatter );
240             this.channel = Objects.requireNonNull( channel );
241             this.last = last;
242 
243         }
244 
245         /**
246          * Creates a reply chain that is initialized to a copy of the state of this chain
247          * but is otherwise functions independently.
248          *
249          * @return The new reply chain.
250          */
251         public ReplyChain copy() {
252 
253             return new ReplyChain( referenceFormatter, channel, last );
254 
255         }
256 
257         /**
258          * Sends the reference message to the given channel.
259          *
260          * @param ch The channel to send to.
261          * @return The sent message ID.
262          */
263         private Mono<Snowflake> sendReference( final MessageChannel ch ) {
264 
265             return ch.createMessage( referenceFormatter.get() )
266                     .map( Message::getId );
267 
268         }
269 
270         /**
271          * Sends reply.
272          *
273          * @param index The reply index.
274          * @param spec The reply specification.
275          * @param channel The target channel.
276          * @return The sent reply.
277          */
278         @SuppressWarnings( "HiddenField" )
279         private Mono<Reply> send( final @NonNegative int index, final ReplySpec spec, 
280                 final MessageChannel channel ) {
281 
282             this.last = this.last.switchIfEmpty( Mono.defer( () -> sendReference( channel ) ) )
283                     .map( spec.toMessage()::withMessageReference )
284                     .flatMap( channel::createMessage )
285                     .map( Message::getId )
286                     .cache();
287 
288             return this.last.map( id -> new MessageReply( 
289                     index, 
290                     id, 
291                     channel.getId(), 
292                     channel.getClient() 
293             ) );
294 
295         }
296 
297         /**
298          * Sends reply.
299          *
300          * @param index The reply index.
301          * @param spec The reply specification.
302          * @return The sent reply.
303          */
304         public Mono<Reply> send( final @NonNegative int index, final ReplySpec spec ) {
305 
306             return channel.flatMap( ch -> send( index, spec, ch ) );
307 
308         }
309 
310     }
