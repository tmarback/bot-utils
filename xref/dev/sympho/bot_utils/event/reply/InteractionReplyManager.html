18  /**
19   * Reply manager for interaction-based commands.
20   *
21   * @version 1.0
22   * @since 1.0
23   */
24  public class InteractionReplyManager extends AbstractReplyManager {
25  
26      /** The title of the reference message. */
27      private final String referenceTitle;
28  
29      /** Generator of the fields for the reference message. */
30      private final Supplier<? extends Iterable<? extends Field>> referenceFields;
31  
32      /** The backing interaction event. */
33      private final DeferrableInteractionEvent event;
34  
35      /** Whether to require that a reply is made before detaching. */
36      private final boolean requireReply;
37  
38      /**
39       * Creates a new message.
40       *
41       * @param referenceTitle The title of the reference message.
42       * @param referenceFields Generator of the fields for the reference message.
43       * @param event The backing interaction event.
44       * @param defaultPrivate Whether replies are private by default.
45       * @param requireReply Whether to require that a reply is made before detaching.
46       */
47      public InteractionReplyManager( 
48              final String referenceTitle,
49              final Supplier<? extends Iterable<? extends Field>> referenceFields,
50              final DeferrableInteractionEvent event, 
51              final boolean defaultPrivate,
52              final boolean requireReply
53      ) {
54  
55          super( defaultPrivate );
56  
57          this.referenceTitle = referenceTitle;
58          this.referenceFields = referenceFields;
59  
60          this.event = event;
61          this.requireReply = requireReply;
62  
63      }
64  
65      @Override
66      protected Mono<Void> doDefer() {
67  
68          return event.deferReply();
69  
70      }
71  
72      @Override
73      protected Mono<Reply> send( final @NonNegative int index, final ReplySpec spec ) {
74  
75          if ( !acked ) { // Initial response
76              if ( index > 0 ) { // Sanity check index
77                  throw new IllegalStateException( "Sending non-intial response while never acked" );
78              }
79  
80              return event.reply( spec.toInteractionReply( defaultPrivate ) ) // Initial response
81                      .thenReturn( new InteractionReply( event ) );
82          } else { // Followup
83              return event.createFollowup( spec.toInteractionFollowup( defaultPrivate ) ) // Followup
84                      .map( m -> new InteractionFollowup( 
85                              index, 
86                              m.getId(), m.getChannelId(), 
87                              event ) 
88                      );
89          }
90  
91      }
92  
93      /**
94       * Formats the reference message.
95       *
96       * @return The formatted message embed.
97       */
98      @SideEffectFree
99      private EmbedCreateSpec formatReference() {
100 
101         return EmbedCreateSpec.builder()
102                 .color( Color.ORANGE )
103                 .title( referenceTitle )
104                 .addAllFields( referenceFields.get() )
105                 .timestamp( event.getInteraction().getId().getTimestamp() )
106                 .build();
107 
108     }
109 
110     /**
111      * Creates the exception for an invalid reply type.
112      *
113      * @param reply The source reply.
114      * @return The exception.
115      */
116     private IllegalStateException invalidReplyType( final Reply reply ) {
117 
118         return new IllegalStateException( 
119                 "Unexpected reply type " + reply.getClass().getName() 
120         );
121 
122     }
123 
124     @Override
125     protected ReplyManager doDetach() {
126 
127         if ( requireReply && replies.isEmpty() ) {
128             throw new IllegalStateException( "Cannot detach before sending a reply" );
129         }
130 
131         final var convertedReplies = replies.stream()
132                 .map( r -> {
133                     if ( r instanceof InteractionReply re ) {
134                         return new InteractionReplyAdapter( re );
135                     } else if ( r instanceof InteractionFollowup f ) {
136                         return new MessageReplyManager.MessageReply( 
137                                 f.index(), 
138                                 f.id(), f.channel(), 
139                                 f.event().getClient()
140                         );
141                     } else {
142                         throw invalidReplyType( r );
143                     }
144                 } )
145                 .toList();
146 
147         final var interaction = event.getInteraction();
148 
149         final var last = convertedReplies.get( convertedReplies.size() - 1 );
150         final Mono<Snowflake> lastId;
151         if ( last instanceof InteractionReplyAdapter r ) {
152             lastId = r.ids.map( i -> i.message() );
153         } else if ( last instanceof MessageReplyManager.MessageReply r ) {
154             lastId = Mono.just( r.id() );
155         } else {
156             throw invalidReplyType( last );
157         }
158 
159         return new MessageReplyManager(
160                 convertedReplies, 
161                 this::formatReference, 
162                 lastId, 
163                 interaction.getChannel(), 
164                 interaction.getGuildId().isPresent()
165                         ? interaction.getUser().getPrivateChannel()
166                         : null, 
167                 defaultPrivate
168         );
169 
170     }
171 
172     @Override
173     public Mono<ReplyManager> detach() {
174 
175         return Mono.fromSupplier( this::doDetach )
176                 .transform( sendLock::guard );
177 
178     }

192     /**
193      * A reply sent as an interaction reply.
194      *
195      * @since 1.0
196      */
197     private static class InteractionReply implements Reply {
198 
199         /** The backing event. */
200         private final DeferrableInteractionEvent event;
201 
202         /** The message IDs (lazy-loaded). */
203         private final Mono<MessageIds> ids;
204 
205         /**
206          * Creates a new instance.
207          *
208          * @param event The backing event.
209          */
210         InteractionReply( final DeferrableInteractionEvent event ) {
211 
212             this.event = event;
213 
214             this.ids = event.getReply()
215                     .map( m -> new MessageIds( m.getId(), m.getChannelId() ) )
216                     .cache(); // IDs are static and can be cached after lazy-loading
217 
218         }
219 
220         @Override
221         public @NonNegative int index() {
222 
223             return 0; // Original reply is always first
224 
225         }
226 
227         @Override
228         public Mono<Message> message() {
229 
230             return event.getReply();
231 
232         }
233 
234         @Override
235         public Mono<Snowflake> messageId() {
236 
237             return ids.map( MessageIds::message );
238 
239         }
240 
241         @Override
242         public Mono<Message> edit( final ReplyEditSpec spec ) {
243 
244             return event.editReply( spec.toInteraction() );
245 
246         }
247 
248         @Override
249         public Mono<Void> delete() {
250 
251             return event.deleteReply();
252 
253         }
254         
255     }

257     /**
258      * A reply sent as an interaction followup.
259      *
260      * @param index The reply index.
261      * @param id The reply ID.
262      * @param channel The reply channel.
263      * @param event The backing event.
264      * @since 1.0
265      */
266     private record InteractionFollowup(
267             @NonNegative int index,
268             Snowflake id,
269             Snowflake channel,
270             DeferrableInteractionEvent event
271     ) implements Reply {
272 
273         @Override
274         public Mono<Message> message() {
275 
276             return event.getClient().getMessageById( channel, id );
277 
278         }
279 
280         @Override
281         public Mono<Snowflake> messageId() {
282 
283             return Mono.just( id );
284 
285         }
286 
287         @Override
288         public Mono<Message> edit( final ReplyEditSpec spec ) {
289 
290             return event.editFollowup( id, spec.toInteraction() );
291 
292         }
293 
294         @Override
295         public Mono<Void> delete() {
296 
297             return event.deleteFollowup( id );
298 
299         }

303     /**
304      * Adapter to convert an interaction reply-based reply into a message-based
305      * reply (mainly interfacing the need to fetch the reply to obtain the ID).
306      *
307      * @since 1.0
308      */
309     private static class InteractionReplyAdapter implements MessageReplyBase {
310 
311         /** The backing client. */
312         private final GatewayDiscordClient client;
313 
314         /** The message IDs. */
315         private final Mono<MessageIds> ids;
316 
317         /**
318          * Creates a new instance.
319          *
320          * @param reply The original reply.
321          */
322         InteractionReplyAdapter( final InteractionReply reply ) {
323 
324             this.client = reply.event.getClient();
325 
326             this.ids = reply.ids;
327             this.ids.subscribe(); // Fetch IDs eagerly before the interaction expires
328             
329         }
330 
331         @Override
332         public @NonNegative int index() {
333 
334             return 0; // Original reply is always first
335 
336         }
337 
338         @Override
339         public Mono<Message> message() {
340 
341             return ids.flatMap( i -> client.getMessageById( i.channel(), i.message() ) );
342 
343         }
344 
345         @Override
346         public Mono<Snowflake> messageId() {
347 
348             return ids.map( MessageIds::message );
349 
350         }
