1   package dev.sympho.bot_utils.event;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import dev.sympho.bot_utils.access.ChannelAccessContext;
6   import dev.sympho.bot_utils.access.ChannelAccessValidator;
7   import dev.sympho.bot_utils.access.Group;
8   import dev.sympho.bot_utils.access.GuildGroup;
9   import discord4j.common.util.Snowflake;
10  import discord4j.core.object.entity.User;
11  import reactor.core.publisher.Mono;
12  
13  /**
14   * The context of an user-triggered event that is bound to a specific channel.
15   *
16   * @version 1.0
17   * @since 1.0
18   */
19  public interface ChannelEventContext 
20          extends EventContext, ChannelAccessContext, ChannelAccessValidator {
21  
22      @Override
23      ChannelAccessValidator validator();
24  
25      /**
26       * @see #validator()
27       * @see ChannelAccessValidator#hasAccess(Group)
28       * @apiNote This is a convenience shortcut for using the {@link #validator() context validator}.
29       *          It is equivalent to {@code validator().hasAccess(group)}.
30       */
31      @Override
32      default Mono<Boolean> hasAccess( final Group group ) {
33  
34          return validator().hasAccess( group );
35  
36      }
37  
38      @Override
39      default Mono<Boolean> hasAccess( final GuildGroup group ) {
40  
41          return validator().hasAccess( group );
42  
43      }
44  
45      /**
46       * @see #validator()
47       * @see ChannelAccessValidator#validate(Group)
48       * @apiNote This is a convenience shortcut for using the {@link #validator() context validator}.
49       *          It is equivalent to {@code validator().validate(group)}.
50       */
51      @Override
52      default Mono<Void> validate( final Group group ) {
53          
54          return validator().validate( group );
55  
56      }
57  
58      /**
59       * Determines whether the given user belongs to the given group in the context of
60       * this event.
61       *
62       * @param user The user to check for.
63       * @param group The group to check for.
64       * @return A Mono that emits {@code true} if the given user belongs to the given
65       *         group under this event context, or {@code false} otherwise.
66       * @apiNote This is a convenience method equivalent to {@code group.belongs(user, this)}.
67       */
68      @SideEffectFree
69      default Mono<Boolean> belongs( final User user, final Group group ) {
70  
71          return group.belongs( user, this )
72                  .defaultIfEmpty( false ); // Just to be safe
73  
74      }
75  
76      @Override
77      default Mono<Boolean> belongs( final User user, final GuildGroup group ) {
78  
79          return belongs( user, ( Group ) group );
80  
81      }
82  
83      /**
84       * Determines whether the given user belongs to the given group in the context of
85       * this event.
86       *
87       * @param user The ID of the user to check for.
88       * @param group The group to check for.
89       * @return A Mono that emits {@code true} if the given user belongs to the given
90       *         group under this event context, or {@code false} otherwise.
91       * @apiNote This is a convenience method equivalent to fetching the user then invoking
92       *          {@code group.belongs(user, this)}.
93       */
94      @SideEffectFree
95      default Mono<Boolean> belongs( final Snowflake user, final Group group ) {
96  
97          return client().getUserById( user ).flatMap( u -> belongs( u, group ) );
98  
99      }
100 
101     @Override
102     default Mono<Boolean> belongs( final Snowflake user, final GuildGroup group ) {
103 
104         return belongs( user, ( Group ) group );
105 
106     }
107     
108 }
